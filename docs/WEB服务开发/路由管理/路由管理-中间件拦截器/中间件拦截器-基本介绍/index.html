дёӯй—ҙд»¶/жӢҰжҲӘеҷЁ-еҹәжң¬д»Ӣз»Қ

GoFrame жҸҗдҫӣдәҶдјҳйӣ…зҡ„дёӯй—ҙд»¶иҜ·жұӮжҺ§еҲ¶ж–№ејҸпјҢиҜҘж–№ејҸд№ҹжҳҜдё»жөҒзҡ„ WebServer жҸҗдҫӣзҡ„иҜ·жұӮжөҒзЁӢжҺ§еҲ¶ж–№ејҸпјҢеҹәдәҺдёӯй—ҙд»¶и®ҫи®ЎеҸҜд»Ҙдёә WebServer жҸҗдҫӣжӣҙзҒөжҙ»ејәеӨ§зҡ„жҸ’д»¶жңәеҲ¶гҖӮз»Ҹе…ёзҡ„дёӯй—ҙд»¶жҙӢи‘ұжЁЎеһӢпјҡ

дёӯй—ҙд»¶е®ҡд№үвҖӢ

дёӯй—ҙд»¶зҡ„е®ҡд№үе’Ңжҷ®йҖҡHTTPжү§иЎҢж–№жі• HandlerFunc дёҖж ·пјҢдҪҶжҳҜеҸҜд»ҘеңЁ Request еҸӮж•°дёӯдҪҝз”Ё Middleware еұһжҖ§еҜ№иұЎжқҘжҺ§еҲ¶иҜ·жұӮжөҒзЁӢгҖӮ

жҲ‘д»¬жӢҝдёҖдёӘи·ЁеҹҹиҜ·жұӮзҡ„дёӯй—ҙд»¶е®ҡд№үжқҘзӨәдҫӢиҜҙжҳҺдёҖдёӢпјҡ

func MiddlewareCORS(r *ghttp.Request) {
	r.Response.CORSDefault()
	r.Middleware.Next()
}

еҸҜд»ҘзңӢеҲ°еңЁиҜҘдёӯй—ҙд»¶дёӯжү§иЎҢе®ҢжҲҗи·ЁеҹҹиҜ·жұӮеӨ„зҗҶзҡ„йҖ»иҫ‘еҗҺпјҢдҪҝз”Ё r.Middleware.Next() ж–№жі•иҝӣдёҖжӯҘжү§иЎҢдёӢдёҖдёӘжөҒзЁӢпјӣеҰӮжһңиҝҷдёӘж—¶еҖҷзӣҙжҺҘйҖҖеҮәдёҚи°ғз”Ё r.Middleware.Next() ж–№жі•зҡ„иҜқпјҢе°ҶдјҡйҖҖеҮәеҗҺз»ӯзҡ„жү§иЎҢжөҒзЁӢпјҲдҫӢеҰӮеҸҜд»Ҙз”ЁдәҺиҜ·жұӮзҡ„йүҙжқғеӨ„зҗҶпјүгҖӮ

дёӯй—ҙд»¶зұ»еһӢвҖӢ

дёӯй—ҙд»¶зҡ„зұ»еһӢеҲҶдёәдёӨз§ҚпјҡеүҚзҪ®дёӯй—ҙд»¶е’ҢеҗҺзҪ®дёӯй—ҙд»¶гҖӮеүҚзҪ®еҚіеңЁи·Ҝз”ұжңҚеҠЎеҮҪж•°и°ғз”Ёд№ӢеүҚи°ғз”ЁпјҢеҗҺзҪ®еҚіеңЁе…¶еҗҺи°ғз”ЁгҖӮ

иҝҷйҮҢзҡ„е»әи®®жқҘеҸӮиҖғдәҺ gRPC зҡ„жӢҰжҲӘеҷЁи®ҫи®ЎпјҢжІЎжңүиҝҮеӨҡзҡ„и·Ҝз”ұжҺ§еҲ¶пјҢд»…еңЁдёҖдёӘең°ж–№еҗҢдёҖдёӘж–№жі•з»ҹдёҖжіЁеҶҢгҖӮеҫҖеҫҖи¶Ҡз®ҖеҚ•пјҢи¶Ҡе®№жҳ“зҗҶи§ЈпјҢд№ҹдҫҝдәҺй•ҝжңҹз»ҙжҠӨгҖӮ

еҲҶз»„и·Ҝз”ұдёӯй—ҙд»¶вҖӢ

еҲҶз»„и·Ҝз”ұдёӯй—ҙд»¶жҳҜз»‘е®ҡеҲ°еҲҶз»„и·Ҝз”ұдёҠзҡ„жңҚеҠЎж–№жі•пјҢдёҚеӯҳеңЁи·Ҝз”ұи§„еҲҷеҢ№й…ҚпјҢеӣ жӯӨеҸӘдјҡжҢүз…§жіЁеҶҢзҡ„е…ҲеҗҺйЎәеәҸжү§иЎҢгҖӮеҸӮиҖғеҗҺз»ӯзӨәдҫӢжҲ–еҰӮдёӢд»Јз Ғзҡ„жү§иЎҢз»“жһңгҖӮ

еұ•ејҖжәҗз Ғ

package main

import (
	"context"
	"github.com/gogf/gf/v2/frame/g"
	"github.com/gogf/gf/v2/net/ghttp"
	"github.com/gogf/gf/v2/os/glog"
)

type HelloReq struct {
	g.Meta `path:"/hello" method:"get"`
}
type HelloRes struct {
}

type Hello struct{}

func (Hello) Say(ctx context.Context, req *HelloReq) (res *HelloRes, err error) {
	glog.Debug(ctx, "дёӯ")
	return
}

func RequestHandle1(r *ghttp.Request) {
	glog.Debug(r.GetCtx(), "еүҚ1")
	r.Middleware.Next()
}

func RequestHandle2(r *ghttp.Request) {
	glog.Debug(r.GetCtx(), "еүҚ2")
	r.Middleware.Next()
}

func RequestHandle3(r *ghttp.Request) {
	glog.Debug(r.GetCtx(), "еүҚ3")
	r.Middleware.Next()
}

func RequestHandle4(r *ghttp.Request) {
	glog.Debug(r.GetCtx(), "еүҚ4")
	r.Middleware.Next()
}

func ResponseHandle1(r *ghttp.Request) {
	r.Middleware.Next()
	glog.Debug(r.GetCtx(), "еҗҺ1")
}

func ResponseHandle2(r *ghttp.Request) {
	r.Middleware.Next()
	glog.Debug(r.GetCtx(), "еҗҺ2")
}

func ResponseHandle3(r *ghttp.Request) {
	r.Middleware.Next()
	glog.Debug(r.GetCtx(), "еҗҺ3")
}

func ResponseHandle4(r *ghttp.Request) {
	r.Middleware.Next()
	glog.Debug(r.GetCtx(), "еҗҺ4")
}

func main() {
	s := g.Server()
	s.Use(ghttp.MiddlewareHandlerResponse)
	s.Group("/", func(group *ghttp.RouterGroup) {
		// еүҚзҪ®дёӯй—ҙд»¶
		group.Middleware(RequestHandle1)
		group.Middleware(RequestHandle2)

		// еҗҺзҪ®дёӯй—ҙд»¶
		group.Middleware(ResponseHandle1)
		group.Middleware(ResponseHandle2)

		group.Group("/sub", func(group *ghttp.RouterGroup) {
			// еүҚзҪ®дёӯй—ҙд»¶
			group.Middleware(RequestHandle3)
			group.Middleware(RequestHandle4)

			// еҗҺзҪ®дёӯй—ҙд»¶
			group.Middleware(ResponseHandle3)
			group.Middleware(ResponseHandle4)

			group.Bind(new(Hello))
		})
	})
	s.Run()
}

жү§иЎҢз»“жһңеҰӮдёӢпјҡ

жү§иЎҢдёӯж–ӯвҖӢ

еңЁ еҲҶз»„и·Ҝз”ұдёӯй—ҙд»¶ дёӯпјҢжҲ‘д»¬еҸҜд»ҘеңЁ еүҚзҪ®дёӯй—ҙд»¶ зҡ„ Next() и°ғз”Ёд№ӢеүҚйҖҡиҝҮ return еҜ№жң¬ж¬ЎиҜ·жұӮиҝӣиЎҢдёӯж–ӯпјҢеңЁжү“ж–ӯеҗҺпјҢ еҗҺз»ӯ жүҖжңү зҡ„ еүҚзҪ®дёӯй—ҙд»¶гҖҒ еҗҢзә§еҸҠеӯҗзә§ зҡ„ еҗҺзҪ®дёӯй—ҙд»¶гҖҒ иҜ·жұӮеӨ„зҗҶж–№жі• е°ҶдёҚдјҡиў«жү§иЎҢгҖӮ

еҰӮдёҠйқўз”ЁжқҘеұ•зӨәеҲҶз»„и·Ҝз”ұдёӯй—ҙд»¶дјҳе…Ҳзә§зҡ„д»Јз Ғдёӯпјҡ

  • еңЁ RequestHandle1 зҡ„ Next() и°ғз”ЁеүҚиҝӣиЎҢдёӯж–ӯпјҢеҲҷеҸӘдјҡжү§иЎҢ RequestHandle1
  • еңЁ RequestHandle2 зҡ„ Next() и°ғз”ЁеүҚиҝӣиЎҢдёӯж–ӯпјҢеҲҷеҸӘдјҡжү§иЎҢ RequestHandle1гҖҒ RequestHandle2
  • еңЁ RequestHandle3 зҡ„ Next() и°ғз”ЁеүҚиҝӣиЎҢдёӯж–ӯпјҢеҲҷдјҡжү§иЎҢ RequestHandle1гҖҒ RequestHandle2гҖҒ RequestHandle3 д»ҘеҸҠ ResponseHandle2гҖҒ ResponseHandle1
  • еңЁ RequestHandle4 зҡ„ Next() и°ғз”ЁеүҚиҝӣиЎҢдёӯж–ӯпјҢеҲҷдјҡжү§иЎҢ RequestHandle1гҖҒ RequestHandle2гҖҒ RequestHandle3гҖҒ RequestHandle4 д»ҘеҸҠ ResponseHandle2гҖҒ ResponseHandle1

дёҠиҝ°зҡ„дёӯж–ӯе®һдҫӢйғҪдёҚдјҡжү§иЎҢ иҜ·жұӮеӨ„зҗҶж–№жі•гҖӮ

йҷӨдәҶеёёи§Ғзҡ„return еңЁ дёӯй—ҙд»¶дёӯз»Ҳз«ҜеҗҺз»ӯеӨ„зҗҶжөҒзЁӢеӨ§ж–№ејҸеӨ–пјҢжЎҶжһ¶иҝҳжҸҗдҫӣдәҶ Exit зӣёе…ізҡ„ж–№жі•жқҘеңЁд»Јз Ғжү§иЎҢеӨ„ејәеҲ¶дёӯж–ӯжү§иЎҢжөҒзЁӢпјҢе…·дҪ“еҸҜеҸӮиҖғз« иҠӮпјҡ ж•°жҚ®иҝ”еӣһ-ExitжҺ§еҲ¶
